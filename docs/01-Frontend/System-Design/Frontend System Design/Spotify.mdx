## D - Data Model (10%)

```typescript
interface Track {
  id: string;
  name: string;
  artist: string;
  album: string;
  duration: number;
  audioUrl: string;
  artwork: string;
}

interface Playlist {
  id: string;
  name: string;
  tracks: Track[];
  owner: string;
  isPublic: boolean;
}
```

## I - Interface (15%)

```typescript
interface AudioPlayerProps {
  currentTrack: Track;
  isPlaying: boolean;
  volume: number;
  onPlay: () => void;
  onPause: () => void;
  onSeek: (time: number) => void;
}

interface StreamingAPI {
  getTrackStream(trackId: string, quality: string): Promise<string>;
  getPlaylist(playlistId: string): Promise<Playlist>;
}
```
